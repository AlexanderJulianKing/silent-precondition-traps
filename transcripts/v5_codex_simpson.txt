from math import sqrt, erf

df=pd.read_csv('experiment_results.csv')
print('overall')
summary=df.groupby('variant')['converted'].agg(['sum','count','mean'])
print(summary)

# two-proportion z test treatment-control, normal approx
control=df[df.variant=='control']['converted']
treat=df[df.variant=='treatment']['converted']
x1=treat.sum(); n1=treat.count(); x0=control.sum(); n0=control.count()
p1=x1/n1; p0=x0/n0
p_pool=(x1+x0)/(n1+n0)
se=sqrt(p_pool*(1-p_pool)*(1/n1+1/n0))
z=(p1-p0)/se
# two-sided p-value using normal cdf
cdf=lambda z: 0.5*(1+erf(z/sqrt(2)))
pval=2*(1-cdf(abs(z)))
print('\\nz_test overall treatment-control')
print({'diff':p1-p0,'z':z,'p_two_sided':pval})
