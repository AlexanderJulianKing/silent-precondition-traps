import csv

total = Decimal('0')
count = 0
excluded = 0
missing = []
by_currency = {}
with open('october_marketplace_orders.csv', newline='') as f:

        if not (row['order_date'].startswith('2026-10-') and row['status'] == 'PAID'):

        key = (row['order_date'], row['currency'])
