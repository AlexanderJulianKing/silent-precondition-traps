import csv

    for r in csv.DictReader(f):

purchases_by_member = defaultdict(list)
with open('purchases_apr_jul.csv', newline='') as f:
    for r in csv.DictReader(f):
        uuid = lookup[(r['store_id'], r['customer_no'])]
        ts = datetime.strptime(r['purchase_ts'], '%Y-%m-%d %H:%M:%S')
        purchases_by_member[uuid].append(ts)

cohort = []
repeaters = []
for uuid, timestamps in purchases_by_member.items():
    dates = sorted(ts.date() for ts in timestamps)
    first = dates[0]
    if first.year == 2026 and first.month == 5:
        cohort.append(uuid)
        if any(1 <= (d - first).days <= 30 for d in dates[1:]):

pct = len(repeaters) / len(cohort) * 100

print('repeaters', len(repeaters))
print('percentage', pct)
